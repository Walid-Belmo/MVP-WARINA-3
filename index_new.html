Basic LED Blink:

void setup() {
  pinMode(13, OUTPUT);
}

void loop() {
  digitalWrite(13, HIGH);
  delay(1000);
  digitalWrite(13, LOW);
  delay(1000);
}

Multiple LEDs:

void setup() {
  pinMode(13, OUTPUT);
  pinMode(12, OUTPUT);
}

void loop() {
  digitalWrite(13, HIGH);
  digitalWrite(12, LOW);
  delay(500);
  digitalWrite(13, LOW);
  digitalWrite(12, HIGH);
  delay(500);
}

Motor Control with PWM:

#include <TimerOne.h>

void setup() {
  Timer1.initialize(20000);    // 20 ms = 50 Hz period
  Timer1.pwm(9, 512);          // 50% duty on pin 9
  Timer1.pwm(10, 256);         // 25% duty on pin 10
}
